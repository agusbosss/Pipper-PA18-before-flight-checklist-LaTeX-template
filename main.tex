\setlength{\columnsep}{12mm}       % gap between left and right column; increase/decrease as desired

% ------------------------------------------------------------
% CHECKLIST TABLE DEFINITION (3 columns)
% ------------------------------------------------------------
% IMPORTANT (technical):
% - \newcolumntype takes a SINGLE LETTER name. We use Q, D, Z.
% - Each p{...} is a fixed-width column; widths are relative to the CURRENT column width (\linewidth).
%
% Column widths (edit these if you want the dots/action aligned differently):
% - Q (item):   64% of column width
% - D (dots):   12% of column width
% - Z (action): 24% of column width
%
% The three should sum to 1.00\linewidth (0.64 + 0.12 + 0.24 = 1.00).
\newcolumntype{Q}{>{\raggedright\arraybackslash}p{0.64\linewidth}} % ITEM text, left-aligned
\newcolumntype{D}{>{\centering\arraybackslash}p{0.12\linewidth}}  % dots column, centered
\newcolumntype{Z}{>{\raggedleft\arraybackslash}p{0.24\linewidth}} % ACTION text, right-aligned

% Environment that starts/ends the table with our three columns.
% @{} removes extra left/right padding at table edges.
\newenvironment{checklist}{%
  \begin{tabular}{@{}QDZ@{}}%
}{%
  \end{tabular}%
}

% One checklist row:
% - #1 = item text (left)
% - #2 = action text (right)
% \dotfill prints a dotted line that stretches in the middle column.
% [-1pt] slightly tightens the row spacing; remove it if you want more vertical space.
\newcommand{\itemline}[2]{%
  \textbf{\MakeUppercase{#1}} & \dotfill & \textbf{\MakeUppercase{#2}}\\[-1pt]%
}

% ------------------------------------------------------------
% HEADERS / TITLES
% ------------------------------------------------------------
% Usage:
%   \hdr{TITLE}         -> normal header with default top spacing
%   \hdr[0pt]{TITLE}    -> header with no extra top spacing (use at top of a column/block)
%
% Everything is uppercased for consistent look (same as your current setup).
\newcommand{\hdr}[2][5pt]{%
  \vspace{#1}%
  {\fontsize{10.3}{11.2}\selectfont\bfseries\MakeUppercase{#2}}\par
  \vspace{2pt}\hrule height 0.4pt\vspace{6pt}%
}

% Page header bar (black band at top with left/right titles):
% - #1: left title
% - #2: right title
% You can change the font size here independently of body text.
\newcommand{\pagehdr}[2]{%
  \begin{tcolorbox}[
    colback=black,     % black background
    colframe=black,    % black border (irrelevant because boxrule=0)
    boxrule=0pt,       % no border line
    arc=0pt,           % square corners (like printed checklists)
    left=6pt,right=6pt,top=4pt,bottom=4pt,
    width=\linewidth   % full text width
  ]
    {\color{white}\fontsize{11.2}{12.4}\selectfont\bfseries\MakeUppercase{#1}}\hfill
    {\color{white}\fontsize{11.2}{12.4}\selectfont\bfseries\MakeUppercase{#2}}
  \end{tcolorbox}
  \vspace{6pt} % spacing after the black bar
}
